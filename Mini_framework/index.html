üíª Example Usage

class Counter extends Component {
  constructor() {
    super();
    this.state = new State({ count: 0 });
  }

  increment() {
    this.state.setState({
      count: this.state.get('count') + 1
    });
  }

  render() {
    return h('div', {},
      h('h3', {}, `Count: ${this.state.get('count')}`),
      h('button', {
        onClick: () => this.increment()
      }, 'Increment')
    );
  }
}

// Mount
const app = new Counter();
app.mount('#app');
